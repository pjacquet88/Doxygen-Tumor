76  {

84  m_bests(1) = std::min(m_bests(1),m_times(1));
85  m_worsts(0) = std::max(m_worsts(0),m_times(0));
86  m_worsts(1) = std::max(m_worsts(1),m_times(1));
87  #endif

119  inline double getCpuTime() const
120  {
